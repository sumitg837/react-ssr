<ol>

    <li> Run below commands <br>     
        <ul>

            <li>
                For Dev<br>
                <ul>
                    <li><code>npm run build:dev</code></li>
                    <li><code>npm run dev</code></li>
                </ul>
            </li>
            <li>
                For Prod<br>
                <ul>
                    <li><code>npm run build</code></li>
                    <li><code>npm start</code></li>
                </ul>
            </li>
        </ul>    
    </li>

</ol>
